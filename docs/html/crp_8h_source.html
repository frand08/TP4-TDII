19 // A macro for placing data into the Code Read Protect (CRP) section,
20 // which is then located at the correct address for the selected MCU
21 // by the automatically generated linker script. The CRP section should
22 // contain a single 32-bit value which is the CRP value. See appropriate
23 // documentation for the MCU to determine CRP values.
24 //
25 // This feature is only available for NXP MCU targets with the Code Read
26 // Protect Feature
27 //
28 // Example:
29 // __CRP const uint32_t CRP_WORD = CRP_NO_CRP ;
30 //
31 #define __CRP __attribute__ ((used,section(".crp")))
